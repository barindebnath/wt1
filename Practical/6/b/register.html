   <script>
      $("form").submit(function(e){
         e.preventDefault();
         $.post(
            "insert.php",
            {
               name: $("#name").val(),
               age: $("#age").val(),
               city: $("#city").val()
            },
            function(result)
            {
               if(result == "success")
               {
                  $("#result").html("Data Inserted Successfully..!");
               }
               else
               {
                  $("#result").html("Error Occured!");
               }
            }
         );
      });
